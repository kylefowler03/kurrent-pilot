  const nodeKey = getNodeKeyFromRequest(req);

  try {
    // Global counts (cheap / non-exact)
    const pingsLastHour = await cheapCount("pings", {
      col: "time_bucket",
      op: "gte",
      val: minBucket,
    });

    // Snapshot-first (canonical 5m field ledger)
    const snapshot5m = await latestSnapshot5m();

    // freshness window: 7 minutes
    const SNAPSHOT_FRESH_MS = 7 * 60 * 1000;

    let snapshot5mAgeMs: number | null = null;
    let snapshot5mFresh = false;

    if (snapshot5m?.bucket_ts) {
      const t = Date.parse(snapshot5m.bucket_ts);
      if (!Number.isNaN(t)) {
        snapshot5mAgeMs = Date.now() - t;
        snapshot5mFresh = snapshot5mAgeMs >= 0 && snapshot5mAgeMs <= SNAPSHOT_FRESH_MS;
      }
    }
    const pingsTotal = await cheapCount("pings");
    const nodeAggLastHour = await cheapCount("node_agg", {
      col: "time_bucket",
      op: "gte",
      val: minBucket,
    });
    const nodeAggTotal = await cheapCount("node_agg");

    const trustNodes = await cheapCount("trust_state", undefined, 100000, "node_key");
    const refRows = await cheapCount("reference_state", undefined, 100000, "time_bucket");

    const devRows = await cheapCount("node_deviation", undefined, 100000, "time_bucket");


    const latestRef = await latestByBucket("reference_state");
    const latestDev = await latestByBucket("node_deviation");

    const latestTrust = await supabase
      .from("trust_state")
      .select("node_key,tau,k_bar,last_bucket,updated_at")
      .neq("node_key", "test_node_001") // remove this line if you prefer
      .order("updated_at", { ascending: false })
      .limit(1)
      .then(({ data, error }) => {
        if (error) throw new Error(`trust_state: ${error.message}`);
        return data && data[0] ? data[0] : null;
      });

    // Node-specific bundle (only if node_key provided)
    let nodeBundle: any = null;

    if (nodeKey) {
      const { data: trustRow, error: trustErr } = await supabase
        .from("trust_state")
        .select("node_key,tau,k_bar,last_bucket,updated_at")
        .eq("node_key", nodeKey)
        .maybeSingle();
      if (trustErr) throw new Error(`trust_state(node): ${trustErr.message}`);

      const { data: devRow, error: devErr } = await supabase
        .from("node_deviation")
        .select("*")
        .eq("node_key", nodeKey)
        .order("time_bucket", { ascending: false })
        .limit(1)
        .maybeSingle();
      if (devErr) throw new Error(`node_deviation(node): ${devErr.message}`);

      // Match reference to the deviation bucket if available,
      // else fall back to latest reference
      let refRow: any = null;
      if (devRow?.time_bucket != null) {
        const { data: r, error: refErr } = await supabase
          .from("reference_state")
          .select("*")
          .eq("time_bucket", devRow.time_bucket)
          .maybeSingle();
        if (refErr) {
          throw new Error(`reference_state(match): ${refErr.message}`);
        }
        refRow = r;
      } else {
        refRow = latestRef;
      }

      nodeBundle = {
        node_key: nodeKey,
        trust_state: trustRow ?? null,
        node_deviation_latest: devRow ?? null,
        reference_for_node: refRow ?? null,
      };
    }

    // Field overlays (bundle into /status so client makes 1 call)
    const [
      cadence,
      coherence,
      vitality,
      expansion,
      summary,
      gate,
      lastGate,
      continuity,
      pilotReadiness,
      ingestHealth,
      gateModeSmoothed,
      trustStateHealth1h,
      gateStability,
      voteReadiness,
    ] = await Promise.all([
      oneRow("field_cadence_score_1h_v1"),
      oneRow("field_coherence_1h_v1"),
      oneRow("field_vitality_1h_v1_mat"),
      oneRow("field_expansion_hints_1h_v1_mat"),
      oneRow("field_summary_1h_v1_mat"),
      oneRow("field_governance_gate_1h_v1_mat"),
      oneRow("field_governance_gate_last_event_v1"),
      oneRow("field_continuity_1h_v1"),
      oneRow("field_pilot_readiness_latest_v1_mat"),
      oneRow("field_ingest_health_v1_mat"),
      oneRow("field_governance_mode_smoothed_latest_v1"),
      oneRow("field_trust_state_health_1h_v1"),
      oneRow("field_governance_gate_stability_1h_v1"),
      oneRow("field_vote_readiness_latest_v1"),
    ]);

    const { data: refreshData, error: refreshErr } = await supabase
      .from("field_refresh_telemetry_v1")
      .select("*")
      .order("refreshed_at", { ascending: false })
      .limit(1);

    if (refreshErr) throw new Error(`field_refresh_telemetry_v1: ${refreshErr.message}`);
    const refreshRow = refreshData?.[0] ?? null;

    // Weather: prefer snapshot if fresh, else fall back to live view
    let weatherRow: any = null;

    if (snapshot5mFresh && snapshot5m) {
      weatherRow = {
        time_bucket: snapshot5m.time_bucket,
        coherence_regime: snapshot5m.coherence_regime,
        n_nodes: snapshot5m.n_nodes,
        w_eff: snapshot5m.w_eff,
        dominance: snapshot5m.dominance,
        sigma_w: snapshot5m.sigma_w,
        ref_k_raw: snapshot5m.ref_k_raw,
        field_weather: snapshot5m.field_weather,
        weather_stability: snapshot5m.weather_stability,
      };
    } else {
      const weatherRes = await supabase
        .from("field_weather_v1")
        .select("*")
        .order("time_bucket", { ascending: false })
        .limit(1);

      weatherRow = weatherRes.data?.[0] ?? null;
    }


    const field = {
      cadence_1h: cadence.ok ? cadence.row : null,
      coherence_1h: coherence.ok ? coherence.row : null,
      vitality_1h: vitality.ok ? vitality.row : null,
      expansion_hints_1h: expansion.ok ? expansion.row : null,
      summary_1h: summary.ok ? summary.row : null, // <-- this is what App.tsx needs

      // Governance gate: prefer snapshot surface if fresh (bucket-consistent with snapshot weather)
      governance_gate_1h: (snapshot5mFresh && snapshot5m)
        ? {
          ok: true,
          now: new Date().toISOString(),

          // snapshot-derived surface
          nodes_active_now: snapshot5m.nodes_active_now,
          vitality_score_0_1: snapshot5m.vitality_score_0_1,
          vitality_regime: snapshot5m.vitality_regime,
          field_dominance: snapshot5m.field_dominance,

          gate_open: snapshot5m.gate_open,
          gate_open_strict: snapshot5m.gate_open_strict,
          gate_open_lenient: snapshot5m.gate_open_lenient,
          gate_reason: snapshot5m.gate_reason,
          recommended_mode: snapshot5m.recommended_mode,

          // helpful alignment info
          field_bucket: snapshot5m.time_bucket,
          bucket_ts: snapshot5m.bucket_ts,
          source: "snapshot_5m",
        }
        : (gate.ok ? gate.row : null),

      governance_gate_last_event: lastGate.ok ? lastGate.row : null,
      continuity_1h: continuity.ok ? continuity.row : null,
      pilot_readiness_latest: pilotReadiness.ok ? pilotReadiness.row : null,
      ingest_health: ingestHealth.ok ? ingestHealth.row : null,
      governance_mode_smoothed: gateModeSmoothed.ok ? gateModeSmoothed.row : null,

      // debug-only; client can ignore
      _errors: {
        cadence: cadence.ok ? null : cadence.error,
        coherence: coherence.ok ? null : coherence.error,
        vitality: vitality.ok ? null : vitality.error,
        expansion: expansion.ok ? null : expansion.error,
        summary: summary.ok ? null : summary.error,
        gate: gate.ok ? null : gate.error,
        lastGate: lastGate.ok ? null : lastGate.error,
        continuity: continuity.ok ? null : continuity.error,
        pilotReadiness: pilotReadiness.ok ? null : pilotReadiness.error,
        ingestHealth: ingestHealth.ok ? null : ingestHealth.error,
      },
    };

    return jsonResponse({
      ok: true,
      now: new Date().toISOString(),
      mech_version: "v0.1.1-trust-core-frozen",
      bucket_ms: bucketMs,
      current_bucket: currentBucket,
      last_hour_min_bucket: minBucket,

      // counts are now non-exact; include a flag so you remember
      counts: {
        pings_last_hour: pingsLastHour.n,
        pings_last_hour_limited: pingsLastHour.limited,
        node_agg_last_hour: nodeAggLastHour.n,
        node_agg_last_hour_limited: nodeAggLastHour.limited,

        trust_state_nodes: trustNodes.n,
        reference_state_rows: refRows.n,
        node_deviation_rows: devRows.n,

        pings_total: pingsTotal.n,
        pings_total_limited: pingsTotal.limited,
        node_agg_total: nodeAggTotal.n,
        node_agg_total_limited: nodeAggTotal.limited,
      },

      latest: {
        reference_state: latestRef,
        node_deviation: latestDev,
        trust_state: latestTrust,
      },

      node: nodeBundle,

      field: {
        ...field,
        trust_state_health_1h: trustStateHealth1h.ok ? trustStateHealth1h.row : null,
        governance_gate_stability_1h: gateStability.ok ? gateStability.row : null,
        vote_readiness: voteReadiness.ok ? voteReadiness.row : null,
        weather: weatherRow,
        snapshot_5m_latest: snapshot5m,
        snapshot_5m_fresh: snapshot5mFresh,
        snapshot_5m_age_ms: snapshot5mAgeMs,
      },

      refresh: refreshRow,
    });

  } catch (err) {
    return jsonResponse({ ok: false, error: (err as Error).message }, 500);
  }
});
